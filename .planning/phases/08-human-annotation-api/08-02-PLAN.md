```python
@classmethod
async def get_annotations(
    cls,
    repo_id: UUID,
    file_path: str | None = None,
    tags: list[str] | None = None,
    limit: int = 10
) -> list["Annotation"]:
    """Fetch annotations for a file context, ranked per DEVBRIDGE_SPEC.md §1287-1291.
    
    Ranking:
    1. Exact file_path match (highest priority)
    2. Tag relevance to query
    3. Upvote count DESC
    """
    query = select(Annotation).where(Annotation.repo_id == repo_id)
    
    if file_path:
        query = query.where(Annotation.file_path == file_path)
    
    if tags:
        query = query.where(Annotation.tags.overlap(tags))
    
    # Order by: exact match first, then upvotes
    query = query.order_by(
        Annotation.file_path == file_path,  # True (1) ranks above False (0)
        Annotation.upvotes.desc()
    ).limit(limit)
    
    return list(await session.execute(query))
```

Also add format method for LLM context:

```python
def format_for_llm(self) -> str:
    """Format annotation for injection into LLM context."""
    lines = [
        f"// @{self.tags} (from line {self.start_line}-{self.end_line})"
    ]
    if self.upvotes > 0:
        lines.append(f"//    ↑ {self.upvotes} helpful votes")
    lines.append(f"{self.comment}")
    return "\n".join(lines)
```
