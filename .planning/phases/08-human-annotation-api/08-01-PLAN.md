```sql
CREATE TABLE annotations (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  repo_id         uuid NOT NULL REFERENCES repositories(id) ON DELETE CASCADE,
  file_path       text NOT NULL,
  start_line      int,
  end_line       int,
  author_id       uuid NOT NULL REFERENCES users(id),
  comment        text NOT NULL,
  tags           text[] DEFAULT '{}',
  embedding      vector(768),
  upvotes       int DEFAULT 0,
  created_at     timestamptz DEFAULT now()
);

CREATE INDEX ON annotations (repo_id, file_path);
```

```python
class Annotation(SQLModel, table=True):
    __tablename__ = "annotations"
    
    id: UUID = Field(default_factory=generate_uuid, primary_key=True)
    repo_id: UUID = Field(foreign_key="repositories.id", nullable=False, index=True)
    file_path: str = Field(nullable=False)
    start_line: int | None = Field(default=None)
    end_line: int | None = Field(default=None)
    author_id: UUID = Field(foreign_key="users.id", nullable=False)
    comment: str = Field(nullable=False)
    tags: list[str] = Field(default_factory=list, sa_column=Column(ARRAY(String), default=[]))
    embedding: list[float] | None = Field(default=None, sa_column=Column(Vector(768)))
    upvotes: int = Field(default=0)
    created_at: datetime = Field(default_factory=datetime.now, sa_column=Column(DateTime(timezone=True), default=datetime.now))
```
